CREATE DATABASE accounts;

CREATE TABLE `accounts`.`users` 
(
    `id` INT NOT NULL AUTO_INCREMENT,
    `first_name` VARCHAR(50) NOT NULL,
     `last_name` VARCHAR(50) NOT NULL,
    `email` VARCHAR(100) NOT NULL,
    `user_type` VARCHAR(10) NOT NULL,
    `login_time` DATETIME,
    
PRIMARY KEY (`id`) 
);
